def get_H_approx_of_WH_X(W, X):
    """ direct solution of min||WH = X|| using numpy Linear Algebra pseudo-inverse """
    WtW = np.dot(W.T, W)
    WtX = np.dot(W.T, X)

    return np.dot(LA.pinv(WtW), WtX)
